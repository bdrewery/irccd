219  template <typename... Args>
220  void call(const std::string &func, Args&&... args) const
221  {
222  Luae::getglobal(*m_process, func);
223 
224  if (Luae::type(*m_process, -1) != LUA_TFUNCTION) {
225  lua_pop(*m_process, 1);
226  } else {
227  auto before = lua_gettop(*m_process);
228  auto after = 0;
229 
230  pushObjects(*m_process, std::forward<Args>(args)...);
231  after = lua_gettop(*m_process);
232 
233  try {
234  Luae::pcall(*m_process, after - before, 0);
235  } catch (std::runtime_error error) {
236  throw Plugin::ErrorException(m_info.name, error.what());
237  }
