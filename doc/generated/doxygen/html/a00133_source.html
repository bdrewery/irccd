224  template <typename... Args>
225  void call(const std::string &func, Args&&... args) const
226  {
227  Luae::getglobal(*m_process, func);
228 
229  if (Luae::type(*m_process, -1) != LUA_TFUNCTION) {
230  lua_pop(*m_process, 1);
231  } else {
232  auto before = lua_gettop(*m_process);
233  auto after = 0;
234 
235  pushObjects(*m_process, std::forward<Args>(args)...);
236  after = lua_gettop(*m_process);
237 
238  try {
239  Luae::pcall(*m_process, after - before, 0);
240  } catch (std::runtime_error error) {
241  throw Plugin::ErrorException(m_info.name, error.what());
242  }
243  }
