922  static Ptr<T> getShared(lua_State *L, int index, const char *meta)
923  {
924  LUAE_STACK_CHECKBEGIN(L);
925 
926  luaL_checktype(L, index, LUA_TUSERDATA);
927  if (!luaL_getmetafield(L, index, FieldName))
928  luaL_error(L, "invalid type cast");
929

931  const char *name = lua_tostring(L, -1);
932  lua_pop(L, 1);
933 
934  bool found(false);
935 
936  if (std::string(name) == std::string(meta)) {
937  found = true;
938  } else {
939  if (!luaL_getmetafield(L, index, FieldParents))
940  luaL_error(L, "invalid type cast");
